int mergesort(int a[],int low,int high,int n)
{
    int mid;
    if( low<high)
       {
          mid=(low+high)/2;
          mergesort(a,low,mid,n);
          mergesort(a,mid+1,high,n);
          merge(a,low,mid,high,n);
       }
     return;
}

int merge(int a[],int low,int mid,int high,int n)
{
    h=low;
    i=low;
    j=mid+1;

    while( h<=mid && j<=high)
    {
        if(a[h]<a[j])
        b[i++]=a[h++];
      else
        b[i++]=a[j++];
    }
      if(h>mid)
    for(k=j; k<=high; k++)
       b[i++]=a[k];

    else
       for(k=h; k<=mid; k++)
       b[i++]=a[k];
       for(k=low; k<=high; k++)
       a[k]=b[k];
 
  return;
}
