#include<stdio.h>
#include<string.h>
#define MAX 50
void sorting(char arr[][MAX],int n)
{
   char t[20];
   int i,pass;
   for(pass=0;pass<=n-2;pass++)
   {
      for(i=0;i<n-2;i++)
      {
        if(stricmp(arr[i],arr[i+1])>0)
        {
           strcpy(t,arr[i]);
           strcpy(arr[i],arr[i+1]);
           strcpy(arr[i+1],t);
        }
      }
    }
}
void main()
{
    char str[MAX][MAX];
