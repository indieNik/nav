#include<conio.h>
#define QUEUE_SIZE 2
int q[QUEUE_SIZE];
int n;
int rear;
int front;
void main()
{
  int x,choice;
 for(;;)
  {
     printf("\n 1: q_insert\n 2: q_delet\n");
     printf(" 3: q_display\n 4: Exit \n");
     printf("Enter the choice:");
     scanf("%d",&choice);
     switch(choice)
     {
        case 1: q_insert();
        break;
        case 2: q_delet();
        break;
        case 3: q_display();
        break;
        default:exit(0);
     }
   }
}
q_insert()
   {
        if (rear==QUEUE_SIZE-1)
        {
            printf("Queue is full");
            return;
        }
        else
        {
            printf("Enter the number to be inserted :");
            scanf("%d",&n);
            rear++;
            q[rear]=n;
            if(front==-1)
            {
                front++;
    }
}
return;
}
q_delet()
{
if(front==-1)
{
printf("Queue is empty ");
return;
}
        else
        {
            if(front==rear)
            front=rear=-1;
            else
            {
                n=q[front];
                printf("Deleted element is %d\n",n);
                front++;
            }
         }
         return;
    }
   q_display()
  {
        int i;
        if(front==-1)
        {
            printf("Queue is empty\n");
            return;
        }
        else
        printf("The contents of queue\n");
        for(i=front;i<=rear;i++)
        {
            printf("%d\n",q[i]);
        }
        return;
    }
